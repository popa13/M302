\question[20]

	\begin{align*}
	y'' - 4y' - 5y = -6e^{-x} .
	\end{align*}

\question[20]
Find the general solution to the following ODE:
	\begin{align*}
	x^2 y'' + xy' - 4y = -6x - 4
	\end{align*}
knowing that $y_1 (x) = x^2$ is a solution to the complementary equation.

\newpage

\question

	\begin{parts}
	\part[5]
	If $y_1$ and $y_2$ are two differentiable functions not identically zero, the Wronkians $W$ of $\{ y_1 , y_2 \}$ is
		\begin{align*}
		W = y_1 y_2' - y_1' y_2 .
		\end{align*}
	Show that if $\{ y_1 , y_2 \}$ is \textbf{not} a set of fundamental solutions for a second order differential equation, then $W = 0$. 
	
	\vfill
	
	\part[5]
	Solve the following IVP:
		\begin{align*}
		y'' + y = 0 , \quad y(0)= 0, y' (0) = 1 .
		\end{align*}
	\vfill
	\end{parts}
